Getting started

Tweet hello world 
<?php
use Social\Twitter;

$twitter = new Twitter\Connection($key, $secret);
$twitter->auth();

echo "<h1>Hi ", $twitter->me()->name, "</h1>";

$twitter->me()->tweet("Hello soci.al world!");
Display all my pictures from Facebook 
<?php
use Social\Facebook;

$facebook = new Facebook\Connection($key, $secret);
$facebook->auth();

echo "<h1>Hi ", $facebook->me()->name, "</h1>";

$result = $facebook->me()->getAlbums()->getPhotos();
foreach ($result as $album=>$photos) {
    echo "<h2>", $album->name, "</h2>";
    foreach ($photos as $photo) {
        echo "<img src='", $photo->picture, "'>";
    }
}
